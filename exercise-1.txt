/*********************
  No 2. Tahun Kabisat
*********************/

  STORE "tahun" with a number
  STORE "kabisat" with false

  IF "tahun" mod 100 equals 0
     IF "tahun" mod 4 equals 0
        SET "kabisat" to true
  ELSE
     IF "tahun" mod 4 equals 0
        SET "kabisat" to true

  DISPLAY kabisat
